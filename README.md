```python
def calc_temps(startdate,enddate):
    t_list=[]
    output = session.query(Measurements).filter(Measurements.date.between(str(startdate),str(enddate)))
    for row in output:
        t_list.append(int(row.tobs))
        tmin = min(t_list)
        tavg = np.mean(t_list)
        tmax = max(t_list)
    return(tmin,tavg,tmax)
```
